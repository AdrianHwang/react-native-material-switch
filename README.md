### Installation
```bash
npm install react-native-material-switch
```

### Usage example
```javascript
var Switch = require('react-native-side-menu');

var Application = React.createClass({
  render: function() {
    return (
      <View>
        <Switch onChangeState={(state)=>{alert(state)}}/>
      </View>
    );
  }
});
```
